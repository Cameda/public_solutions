# Change shutdownGracePeriod in Kubelet config
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: kubelet-config-updater
  namespace: default
  labels:
    env: test
    cam: ds-kubelet-config-updater
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      name: kubelet-config-updater
  template:
    metadata:
      labels:
        name: kubelet-config-updater
    spec:
      containers:
      - name: kubelet-config-updater
        image: busybox
        imagePullPolicy: IfNotPresent
        resources:
          limits:
            memory: 70Mi
          requests:
            cpu: 50m
            memory: 50Mi
        command:
        - /bin/sh
        - -c
        - |
          while true; do
            echo "Updating kubelet configuration..."
            sed -i 's/shutdownGracePeriod: .*/shutdownGracePeriod: 60s/' /host/home/kubernetes/kubelet-config.yaml
            echo "Configuration updated. Restarting kubelet..."
            systemctl restart kubelet
            sleep 3600
          done
        volumeMounts:
        - name: kubelet-config
          mountPath: /host/home/kubernetes
      hostNetwork: true
      restartPolicy: Always
      hostname: kubelet-config-updater
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
      - name: kubelet-config
        hostPath:
          path: /home/kubernetes
```
