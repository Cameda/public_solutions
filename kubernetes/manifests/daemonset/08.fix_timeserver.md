# Delete one of the timeserver from timesyncd.conf
```
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: fix-timeserver
  namespace: default
  labels:
    k8s-app: fix-timeserver
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      name: fix-timeserver
  template:
    metadata:
      labels:
        name: fix-timeserver
    spec:
      containers:
      - name: fix-timeserver
        image: busybox
        imagePullPolicy: IfNotPresent
        resources:
          limits:
            memory: 70Mi
          requests:
            cpu: 50m
            memory: 50Mi
        command:
        - /bin/sh
        - -c
        - |
            echo "Updating ntp configuration in systemd-timesyncd..."
            sed -i 's/clock.isc.org //g' /host/etc/systemd/timesyncd.conf
            sleep infinity
        securityContext:
          privileged: true
        volumeMounts:
        - name: fix-timeserver
          mountPath: /host/etc/systemd
      hostNetwork: true
      restartPolicy: Always
      hostname: fix-timeserver
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
      - name: fix-timeserver
        hostPath:
          path: /etc/systemd
EOF
```
