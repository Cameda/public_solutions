# Поправим mtu на eth0 на нодах кластера
```
cat <<EOF | kubectl apply -f -
apiVersion: "apps/v1"
kind: DaemonSet
metadata:
  name: mtu-fix
  namespace: default
  labels:
    k8s-app: mtu-fix
spec:
  selector:
    matchLabels:
      k8s-app: mtu-fix
  template:
    metadata:
      labels:
        k8s-app: mtu-fix
    spec:
      containers:
      - name: mtu-fix
        image: cr.yandex/yc/mk8s-openssl:stable
        imagePullPolicy: IfNotPresent
        command:
          - bash
          - -c
          - |
            chroot /host /bin/bash -c "ip link set dev eth0 mtu 1450 &&  sleep infinity"
        securityContext:
          privileged: true
        resources:
          limits:
            memory: 50Mi
          requests:
            cpu: 70m
            memory: 50Mi
        volumeMounts:
          - mountPath: /host
            name: host-namespace
      hostPID: true
      hostIPC: true
      hostNetwork: true
      restartPolicy: Always
      hostname: mtu-fix
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
        - name: host-namespace
          hostPath:
            path: /
            type: Directory
EOF
```
