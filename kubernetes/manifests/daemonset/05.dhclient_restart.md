# DHCP client restart
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: dhclient-restart
  labels:
    environment: test
    k8s-app: dhclient
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      app: dhclient
  template:
    metadata:
      labels:
        app: dhclient
    spec:
      containers:
      - name: dhclient
        image: busybox
        imagePullPolicy: IfNotPresent
        resources:
          limits:
            memory: 70Mi
          requests:
            cpu: 50m
            memory: 50Mi
        command:
        - /bin/sh
        - -c
        - |
          chroot /mnt/cameda /bin/sh -c "dhclient -v -r && dhclient && sleep infinity"
        securityContext:
          privileged: true
        volumeMounts:
        - name: rootdir
          mountPath: /mnt/cameda
      hostPID: true
      hostIPC: true
      hostNetwork: true
      hostname: dhclient
      restartPolicy: Always
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
      - name: rootdir
        hostPath:
          path: /
          type: Directory
```
