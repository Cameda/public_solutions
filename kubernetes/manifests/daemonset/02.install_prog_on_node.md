# Установка программ на ноды кластера
```
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: install-prog
  namespace: default
  labels:
    k8s-app: install-prog
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      app: install-prog
  template:
    metadata:
      labels:
        app: install-prog
    spec:
      containers:
      - name: install-prog
        image: busybox
        imagePullPolicy: IfNotPresent
        resources:
          limits:
            memory: 150Mi
          requests:
            cpu: 150m
            memory: 150Mi
        command:
          - /bin/sh
          - -c
          - |
            chroot /mnt/cameda /bin/sh -c "sudo apt -y update && sudo apt -y install htop dstat traceroute && sleep infinity"
        securityContext:
          privileged: true
        volumeMounts:
        - name: rootdir
          mountPath: /mnt/cameda
      hostPID: true
      hostNetwork: true
      restartPolicy: Always
      hostname: install-prog
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
      - name: rootdir
        hostPath:
          path: /
          type: Directory
EOF
```
