# Установка программ на ноды кластера
```
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: install-prog
  namespace: default
  labels:
    cam: install-prog
    cam: ds
    cam: prog
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      app: install-prog
  template:
    metadata:
      labels:
        app: install-prog
    spec:
      containers:
      - name: install-prog
        image: busybox
        imagePullPolicy: IfNotPresent
        securityContext:
          privileged: true
        resources:
          requests:
            cpu: "50m"
            memory: "100Mi"
          limits:
            memory: "150Mi"
        volumeMounts:
        - name: rootdir
          mountPath: /mnt/cameda
        command:
          - /bin/sh
          - -c
          - |
            chroot /mnt/cameda /bin/sh -c "sudo apt -y update && sleep 50 && sudo apt -y install dstat traceroute && sleep infinity"
      hostNetwork: true
      restartPolicy: Always
      nodeSelector:
        kubernetes.io/os: linux
      volumes:
      - name: rootdir
        hostPath:
          path: /
          type: Directory
      tolerations:
      - key: "cam"
        operator: "Exists"
        effect: "NoSchedule"
      - key: "cam"
        operator: "Exists"
        effect: "NoExecute"
```
