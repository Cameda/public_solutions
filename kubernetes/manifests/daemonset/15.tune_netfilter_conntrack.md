# Установка параметров Conntrack на нодах
```
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: ch-conntrack
  labels:
    name: ch-conntrack
spec:
  selector:
    matchLabels:
      name: ch-conntrack
  template:
    metadata:
      labels:
        name: ch-conntrack
    spec:
      hostNetwork: true
      containers:
      - name: ch-conntrack
        image: nginx:alpine-perl
        imagePullPolicy: IfNotPresent
        command:
        - "/bin/sh"
        args:
        - "-c"
        - while :; do  time=$(date);done
        imagePullPolicy: IfNotPresent
        lifecycle:
          postStart:
            exec:
              command:
              - sysctl
              - "-w"
              - net.netfilter.nf_conntrack_max=1250000
              - net.netfilter.nf_conntrack_tcp_timeout_close_wait=300
        securityContext:
          privileged: true
      imagePullSecrets:
      - name: default-secret
      restartPolicy: Always
      hostname: ch-conntrack
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
EOF
```
