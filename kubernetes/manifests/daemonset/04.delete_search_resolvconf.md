# Удалим строчку search из /etc/resolv.conf на нодах кластера
```
cat <<EOF | kubectl apply -f -
apiVersion: "apps/v1"
kind: DaemonSet
metadata:
  name: delete-search-resolv
  namespace: default
  labels:
    k8s-app: delete-search
    env: test
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      app: delete-search
  template:
    metadata:
      labels:
        app: delete-search
    spec:
      containers:
      - name: delete-search
        image: amouat/network-utils
        imagePullPolicy: IfNotPresent
        command:
          - bash
          - -c
          - |
            chroot /host /bin/sh -c "sed '/search/d' /etc/resolv.conf > temp_file && cat temp_file > /etc/resolv.conf && rm -rf temp_file && sleep infinity"
        securityContext:
          privileged: true
        resources:
          limits:
            memory: 70Mi
          requests:
            cpu: 50m
            memory: 50Mi
        volumeMounts:
          - mountPath: /host
            name: host-namespace
      hostPID: true
      hostIPC: true
      hostNetwork: true
      restartPolicy: Always
      hostname: delete-search
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists 
      volumes:
        - name: host-namespace
          hostPath:
            path: /
            type: Directory
EOF
```
