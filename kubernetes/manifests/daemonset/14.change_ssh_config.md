# Turn off ssh password login
```
cat <<EOF | kubectl apply -f -
apiVersion: apps/v1
kind: DaemonSet
metadata:
  name: ssh-pass-login
  namespace: default
  labels:
    k8s-app: ssh-pass-login
  annotations:
    author: cameda
spec:
  selector:
    matchLabels:
      name: ssh-pass-login
  template:
    metadata:
      labels:
        name: ssh-pass-login
    spec:
      containers:
      - name: ssh-pass-login
        image: busybox
        imagePullPolicy: IfNotPresent
        resources:
          limits:
            memory: 70Mi
          requests:
            cpu: 50m
            memory: 50Mi
        command:
        - /bin/sh
        - -c
        - |
            echo "Updating sshd_config in /etc/ssh..."
            sudo sed -i 's/#PasswordAuthentication yes/PasswordAuthentication no/g' /host/etc/ssh/sshd_config
            sudo systemctl restart sshd
            sleep infinity
        securityContext:
          privileged: true
        volumeMounts:
        - name: ssh-pass-login
          mountPath: /host/etc/ssh
      hostNetwork: true
      restartPolicy: Always
      hostname: ssh-pass-login
      nodeSelector:
        kubernetes.io/os: linux
      tolerations:
      - effect: NoSchedule
        operator: Exists
      volumes:
      - name: ssh-pass-login
        hostPath:
          path: /etc/ssh
EOF
```
