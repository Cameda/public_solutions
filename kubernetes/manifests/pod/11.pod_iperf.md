# Pod with iperf3 server check UDP
```
cat <<EOF | kubectl apply -f -
apiVersion: v1
kind: Pod
metadata:
  name: cam-pod-iperf
  namespace: default
  labels:
    cam: iperf
  annotations:
    author: cameda
spec:
  containers:
  - name: iperf
    image: quay.io/wcaban/net-toolbox:latest
    imagePullPolicy: IfNotPresent
    ports:
    - containerPort: 5201
      hostPort: 35001
      protocol: UDP
    resources:
      requests:
        cpu: "20m"
        memory: "20Mi"
      limits:
        memory: "120Mi"
    command:
    - iperf3
      - "-s"
      - "-B"
      - "0.0.0.0"
      - "-V"
      - "-i"
      - "1"
      - --forceflush
  restartPolicy: OnFailure
  hostname: iperf
  hostNetwork: false
  nodeSelector:
    kubernetes.io/os: linux
EOF
```
